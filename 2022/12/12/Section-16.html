import random

def diceRoll():
    x=0
    diceOne = 0
    diceTwo = 0
    diceThree = 0
    diceFour = 0
    diceFive = 0
    diceSix = 0
    while x < 1000:
        diceNum = random.randrange(1,7)

        if diceNum == 1:
            diceOne = diceOne + 1

        if diceNum == 2:
            diceTwo = diceTwo + 1

        if diceNum == 3:
            diceThree = diceThree + 1

        if diceNum == 4:
            diceFour = diceFour + 1

        if diceNum == 5:
            diceFive = diceFive + 1

        if diceNum == 6:
            diceSix = diceSix + 1
        
        x = x+1
    OneP = diceOne/10

    print("Six:",diceSix,"times or",SixP,"%")
